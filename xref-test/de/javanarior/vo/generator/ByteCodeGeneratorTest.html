71      public void testImplementedInterfaces() {
72          Assert.assertEquals(ByteCodeGenerator.implementedInterfaces(IntType.class)[0],
73                          IntType.class.getName().replace('.', '/'));
74      }

76      public void testParentClassName() {
77          Assert.assertEquals(ByteCodeGenerator.parentClassName(IntWrapper.class),
78                          IntWrapper.class.getName().replace('.', '/'));
79      }
80  
81      public void testImplementationClassName() {
82          String implementedInterfaceName = IntType.class.getName().replace(".", "/");
83          String generatedNamePart = "$VO$" + Integer.toHexString(implementedInterfaceName.hashCode());
84          Assert.assertEquals(ByteCodeGenerator.implementationClassName(IntType.class), implementedInterfaceName
85                          + generatedNamePart);
86          Assert.assertTrue(ByteCodeGenerator.implementationClassName(IntType.class).matches(
87                          implementedInterfaceName + "\\$VO\\$[0-9a-f]+"));
88      }
89  
90      public void testImplementationClassNameIsEqualForEqualTypes() {
91          String implementationClassName1 = ByteCodeGenerator.implementationClassName(IntType.class);
92          String implementationClassName2 = ByteCodeGenerator.implementationClassName(IntType.class);
93          Assert.assertEquals(implementationClassName1, implementationClassName2);
94      }
95  
96      public void testInterfaceName() {
97          Assert.assertEquals(ByteCodeGenerator.interfaceName(IntType.class), IntType.class.getName().replace(".", "/"));
98      }

145     @Test(dataProvider = "getWrapper")
146     public void testGenerate(Class<?> interfaze, Class<? extends Comparable<?>> technicalType,
147                     Class<? extends AbstractValue<?, ?>> superClass, Object value)
148                     throws Exception {
149         ByteCodeContainer generatedClass = ByteCodeGenerator.generate(interfaze, technicalType, superClass);
150         ByteCodeClassLoader classLoader = ByteCodeClassLoader.getClassLoader();
151         Class<?> load = classLoader.load(generatedClass);
152         Assert.assertNotNull(load);
153         Value<?, ?> valueObject = (Value<?, ?>)Invoker.invokeConstructor(load, technicalType, value);
154         Assert.assertEquals(valueObject.getValue(), value);
155     }
156 
157     public void testGenerateWithClassAsType() {
158         try {
159             ByteCodeGenerator.generate(IntValue.class, Integer.TYPE, IntWrapper.class);
160         } catch (IllegalArgumentException exception) {
161             Assert.assertEquals(exception.getMessage(),
162                             "Could not generate implementation for class " + IntValue.class.getName()
163                                             + ". Please provide interface");
164         }
165     }
