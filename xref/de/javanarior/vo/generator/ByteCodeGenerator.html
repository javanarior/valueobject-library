16  package de.javanarior.vo.generator;
17  
18  import java.lang.reflect.Modifier;
19  
20  import org.objectweb.asm.ClassWriter;
21  import org.objectweb.asm.Label;
22  import org.objectweb.asm.MethodVisitor;
23  import org.objectweb.asm.Opcodes;
24  import org.objectweb.asm.Type;
25  
26  import de.javanarior.utils.lang.ByteCodeContainer;
27  import de.javanarior.vo.types.AbstractValue;
28  
29  /**
30   * Generate a implementation of a value object interface.
31   */
32  public final class ByteCodeGenerator {
33  
34      private static final int OBJECT_PRIMITIVE_SIZE = 2;
35      private static final int DOUBLE_SIZE = 3;
36      private static final String GENERATED_VO_INDICATOR = "$VO$";
37      private static final String CONSTRUCTOR = "<init>";
38  
39      private ByteCodeGenerator() {
40      }
41  
42      static String generateImplementationClassName(Class<?> valueType) {
43          String name = Type.getInternalName(valueType);
44          return name + GENERATED_VO_INDICATOR + Integer.toHexString(name.hashCode());
45      }
46  
47      static String implementationClassName(Class<?> valueType) {
48          return ByteCodeGenerator.generateImplementationClassName(valueType);
49      }
50  
51      static String interfaceName(Class<?> valueType) {
52          return Type.getInternalName(valueType);
53      }
54  
55      static String binaryClassName(Class<?> valueType) {
56          return implementationClassName(valueType).replace('/', '.');
57      }
58  
59      static String parentClassName(Class<?> superClass) {
60          return Type.getInternalName(superClass);
61      }
62  
63      static String[] implementedInterfaces(Class<?> valueType) {
64          return new String[] { interfaceName(valueType) };
65      }
66  
67      static String methodDescriptor(Class<?> type) {
68          return "(" + getType(type) + ")V";
69      }
70  
71      static String addTypeSignature(String type) {
72          return "L" + type + ";";
73      }
74  
75      static int getILOADOpCode(Class<?> type) {
76          return Type.getType(type).getOpcode(Opcodes.ILOAD);
77      }
78  
79      static int getStackSize(String desc) {
80          switch (desc.charAt(0)) {
81              case 'D':
82              case 'J':
83                  return DOUBLE_SIZE;
84              default:
85                  return OBJECT_PRIMITIVE_SIZE;
86          }
87      }
88  
89      static String addTypeDiscriptor(Class<?> type) {
90          return Type.getDescriptor(type);
91      }
92  
93      static String getType(Class<?> type) {
94          return Type.getType(type).toString();
95      }
96  
97      /**
98       * Generate a implementation of a value object.
99       *
100      * @param valueType
101      *            - value object type
102      * @param technicalType
103      *            - to which the value object is mapped
104      * @param wrapperClass
105      *            - abstract wrapper class, correspond to the technical type
106      * @return class name and byte code in a container
107      */
108     /* CHECKSTYLE:OFF */
109     public static ByteCodeContainer generate(Class<?> valueType, Class<? extends Comparable<?>> technicalType,
110                     @SuppressWarnings("rawtypes") Class<? extends AbstractValue> wrapperClass) {
111         /* CHECKSTYLE:ON */
112 
113         if (!isInterface(valueType)) {
114             throw new IllegalArgumentException("Could not generate implementation for class " + valueType.getName()
115                             + ". Please provide interface");
116         }
117 
118         ClassWriter classWriter = new ClassWriter(0);
119         MethodVisitor methodVisitor;
120 
121         classWriter.visit(Opcodes.V1_7, Opcodes.ACC_PUBLIC + Opcodes.ACC_SUPER, implementationClassName(valueType), "L"
122                         + parentClassName(wrapperClass) + "<"
123                         + addTypeDiscriptor(valueType) + ">;" + addTypeDiscriptor(valueType),
124                         parentClassName(wrapperClass), implementedInterfaces(valueType));
125 
126         methodVisitor = classWriter.visitMethod(Opcodes.ACC_PUBLIC, CONSTRUCTOR, methodDescriptor(technicalType), null,
127                         null);
128         methodVisitor.visitCode();
129         Label label0 = new Label();
130         methodVisitor.visitLabel(label0);
131         /* CHECKSTYLE:OFF */
132         methodVisitor.visitLineNumber(8, label0);
133         /* CHECKSTYLE:ON */
134         methodVisitor.visitVarInsn(Opcodes.ALOAD, 0);
135         methodVisitor.visitVarInsn(getILOADOpCode(technicalType), 1);
136         methodVisitor.visitMethodInsn(Opcodes.INVOKESPECIAL, parentClassName(wrapperClass), CONSTRUCTOR,
137                         methodDescriptor(technicalType),
138                         false);
139         Label label1 = new Label();
140         methodVisitor.visitLabel(label1);
141         /* CHECKSTYLE:OFF */
142         methodVisitor.visitLineNumber(9, label1);
143         /* CHECKSTYLE:ON */
144         methodVisitor.visitInsn(Opcodes.RETURN);
145         Label label2 = new Label();
146         methodVisitor.visitLabel(label2);
147         methodVisitor.visitLocalVariable("this", addTypeSignature(implementationClassName(valueType)), null, label0,
148                         label2, 0);
149         methodVisitor.visitLocalVariable("value", getType(technicalType), null, label0, label2, 1);
150         int stackSize = getStackSize(Type.getDescriptor(technicalType));
151         methodVisitor.visitMaxs(stackSize, stackSize);
152         methodVisitor.visitEnd();
153         classWriter.visitEnd();
154 
155         return new ByteCodeContainer(binaryClassName(valueType), classWriter.toByteArray());
156     }
157 
158     private static boolean isInterface(Class<?> valueType) {
159         return Modifier.isInterface(valueType.getModifiers());
160     }
161 
162 }
